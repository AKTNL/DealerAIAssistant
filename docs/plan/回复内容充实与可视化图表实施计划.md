**目标：** 丰富 AI 回复内容，增加更深入的分析，并在所有 6 个业务场景的 HTML 表格旁边添加 Mermaid 图表，使用中文段落标题。

**技术栈：** Java 21（后端），Vue 3 + markdown-it + Mermaid 11.x（前端）

markdown.renderer.rules.fence = (tokens, index, options, env, self) => {
  const token = tokens[index];
  const info = token.info ? markdown.utils.unescapeAll(token.info).trim() : "";
  const langName = info ? escapeHtml(info.split(/\s+/g)[0]) : "";

  if (langName === "mermaid") {
    return `<pre class="mermaid-chart">${escapeHtml(token.content)}</pre>`;
  }

  const highlighted = options.highlight
    ? options.highlight(token.content, info, "")
    : escapeHtml(token.content);

  return `<pre class="hljs"><code class="hljs${langName ? ` language-${langName}` : ""}">${highlighted}</code></pre>`;
};

async function renderMermaidCharts(root) {
  const charts = root.querySelectorAll(".mermaid-chart:not(.mermaid-rendered)");

  for (const chart of charts) {
    try {
      const content = chart.textContent;

private String sanitizeMermaidLabel(String name) {
    if (name == null || name.isBlank()) {
        return "";
    }
    String cleaned = name
            .replaceAll("[（(][^)）]*[)）]", "")
            .replaceAll("[\"'\\[\\]{}]", "")
            .trim();
    if (cleaned.isEmpty()) {
        return "";
    }
    if (cleaned.length() > 10) {
        cleaned = cleaned.substring(0, 9) + "…";
    }
    return cleaned;
}

private String buildMermaidXyChart(
        String title,
        String xAxisLabel,
        String yAxisLabel,
        List<String> labels,
        List<Double> barValues,
        Double averageLineValue
) {
    StringBuilder sb = new StringBuilder();
    sb.append("```mermaid\n");
    sb.append("xychart-beta\n");
    sb.append("    title \"").append(sanitizeMermaidLabel(title)).append("\"\n");
    sb.append("    x-axis [");
    sb.append(labels.stream()
            .map(l -> "\"" + sanitizeMermaidLabel(l) + "\"")
            .collect(Collectors.joining(", ")));
    sb.append("]\n");
    sb.append("    y-axis \"").append(sanitizeMermaidLabel(yAxisLabel)).append("\"");
    if (averageLineValue != null && !barValues.isEmpty()) {
        double maxVal = barValues.stream().mapToDouble(Double::doubleValue).max().orElse(100);
        sb.append(" 0 --> ").append((int) Math.ceil(Math.max(maxVal, 100) * 1.1));
    }
    sb.append("\n");
    sb.append("    bar [");
    sb.append(barValues.stream()
            .map(v -> String.format(Locale.US, "%.1f", v))
            .collect(Collectors.joining(", ")));
    sb.append("]\n");
    if (averageLineValue != null) {
        String avgStr = String.format(Locale.US, "%.1f", averageLineValue);
        sb.append("    line [");
        for (int i = 0; i < labels.size(); i++) {
            if (i > 0) sb.append(", ");
            sb.append(avgStr);
        }
        sb.append("]\n");
    }
    sb.append("```");
    return sb.toString();
}

private String buildFallbackBars(List<String> labels, List<Double> values, double maxValue) {
    StringBuilder sb = new StringBuilder();
    sb.append("<pre class=\"mermaid-chart-fallback\" style=\"display:none;\">\n");
    int maxBlocks = 20;
    for (int i = 0; i < labels.size(); i++) {
        String label = sanitizeMermaidLabel(labels.get(i));
        double val = values.get(i);
        int blocks = (int) Math.round(val / maxValue * maxBlocks);
        sb.append("▌".repeat(Math.max(0, blocks)));
        sb.append(" ".repeat(Math.max(0, maxBlocks - blocks + 1)));
        sb.append(label).append("  ").append(String.format(Locale.US, "%.1f%%", val)).append("\n");
    }
    sb.append("</pre>");
    return sb.toString();
}
```

    List<String> allPoints = Stream.concat(analysisPoints.stream(), recommendationPoints.stream()).toList();
    return buildEnrichedReply(language,
            isZh(language) ? "- " + conclusion : "- " + conclusion,
            dataRows, null, null, List.of(), allPoints, followUpQuestions
    );
}

private boolean isZh(String language) { return "zh".equals(language); }
```
